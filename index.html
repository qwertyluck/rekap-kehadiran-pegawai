function doGet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheetPegawai = ss.getSheetByName("Pegawai"); var sheetAbsensi = ss.getSheetByName("Absensi"); var result = { pegawai: [], absensi: [] }; // 1. Mengambil data Pegawai if (sheetPegawai) { var dataPeg = sheetPegawai.getDataRange().getValues(); for (var i = 1; i < dataPeg.length; i++) { if (dataPeg[i][0] != "") { // Mengambil NIP (Kolom A) dan Nama (Kolom B) result.pegawai.push({ nip: dataPeg[i][0].toString(), n: dataPeg[i][1].toString() }); } } } // 2. Mengambil data Absensi if (sheetAbsensi) { var dataAbs = sheetAbsensi.getDataRange().getValues(); for (var j = 1; j < dataAbs.length; j++) { if (dataAbs[j][0] != "") { // Mengambil NIP, Tanggal, Status result.absensi.push({ nip: dataAbs[j][0].toString(), tanggal: dataAbs[j][1].toString(), status: dataAbs[j][2].toString() }); } } } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON) .setHeader("Access-Control-Allow-Origin", "*"); } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Absensi"); var params = JSON.parse(e.postData.contents); var data = sheet.getDataRange().getValues(); var updated = false; // Cek jika data absensi di tanggal tersebut sudah ada, maka update/timpa for (var i = 1; i < data.length; i++) { if (data[i][0].toString() == params.nip.toString() && data[i][1].toString() == params.tanggal.toString()) { sheet.getRange(i + 1, 3).setValue(params.status); updated = true; break; } } // Jika belum ada, buat baris baru if (!updated) { sheet.appendRow([params.nip, params.tanggal, params.status]); } return ContentService.createTextOutput("Sukses").setMimeType(ContentService.MimeType.TEXT) .setHeader("Access-Control-Allow-Origin", "*"); }
